This directory contains sample outputs generated by scripts of the projet

cpe_list.txt :
	scripts/pnm-extract-cpe.py -i xml_examples/S2_NAT.xml -o sample_results/cpe_list.txt

ports-hits.html :
	scripts/pnm-ports-hits sample_results/ xml_examples/better_example.xml

ports-maps.html :
	scripts/pnm-summarize.py -i xml_examples/better_example.xml | scripts/pnm-ports-map sample_results/

vendors-hits.html :
	scripts/pnm-extract-mac.py -i xml_examples/better_example.xml -g | scripts/pnm-vendors-hits sample_results/

words-cloud.html :
	scripts/pnm-extract-words.py -i xml_examples/example_with_cpe_tags.xml | scripts/pnm-words-cloud out/
